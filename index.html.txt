<script>

let numeroAtual = BigInt("61229000000000000");
let ultimoNumero = "";

function gerarNumero(){

    let numero = numeroAtual.toString();

    if(numero.length < 17){
        numero = numero.padStart(17,"0");
    }

    document.getElementById("lista").innerHTML += numero + "<br>";

    ultimoNumero = numero;

    numeroAtual++;
}

function copiarNumero(){

    if(ultimoNumero === ""){
        alert("Gere um número primeiro");
        return;
    }

    navigator.clipboard.writeText(ultimoNumero);

    alert("Número copiado: " + ultimoNumero);
}
